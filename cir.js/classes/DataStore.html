• get([key])

    データを取り出します。

    store.set('cir', 'Cool is Right.'); // set
var text = store.get('cir'); // Cool is Right.
var obj = store.get(); // {cir: 'Cool is Right.'}

  • remove(key)

    データを削除します。

    store.set('cir', 'Cool is Right.'); // set
if (store.remove('cir')) {
    alert('remove.');
}

  • reset()

    データを全て削除します。

    store.set('c', 'Cool'); // set
store.set('i', ' is '); // set
store.set('r', 'Right.'); // set
store.reset();

var c = store.get('c'); // undefind
var i = store.get('i'); // undefind
var r = store.get('r'); // undefind

  • set(key, val)

    データを保存します。

    store.set('cir', 'Cool is Right.'); // String
store.set('obj', {
    c: 'Cool',
    i: ' is ',
    r: 'Right.'
}); // Object
store.set('num', 123); // Number
