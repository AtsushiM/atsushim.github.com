• addChild(instance)

    Observer、もしくはObserverを継承したクラスのインスタンスを子として登録します。登録された子はイベント伝播の対象になります。

    var observer2 = new C.Observer();
observer.addChild(observer2);

  • bubble(event, [vars])

    fireメソッドのエイリアスです。イベントはバブリングで伝播されます。

    observer.on('jsonloaded', function(vars) {
    // write code.
    console.log(vars); // {cir: 'Cool is Right.'}
});

var observer2 = new C.Observer();

observer2.on('jsonloaded', function() {
    // write code.
});

observer.addChild(observer2);

observer2.bubble('jsonloaded', {
    cir: 'Cool is Right.'
});

  • capture(event, [vars])

    イベントをキャプチャリングで伝播します。

    observer.on('jsonloaded', function(vars) {
    // write code.
    console.log(vars); // {cir: 'Cool is Right.'}
});

var observer2 = new C.Observer();

observer2.on('jsonloaded', function() {
    // write code.
});

observer.addChild(observer2);

observer.capture('jsonloaded', {
    cir: 'Cool is Right.'
});

  • fire(event, [vars])

    onメソッドで登録されたイベントを実行します。イベントはバブリングで伝播されます。

    observer.on('jsonloaded', function(vars) {
    // write code.
    console.log(vars); // {cir: 'Cool is Right.'}
});

observer.fire('jsonloaded', {
    cir: 'Cool is Right.'
});

  • off(key, [func])

    登録されたイベントを削除します。

    var func1 = function() {
    console.log('func1');
};
var func2 = function() {
    console.log('func2');
};

observer.on('jsonloaded', func1);
observer.on('jsonloaded', func2);

observer.off('jsonloaded', func1);

observer.fire('jsonloaded'); // exe func2.

observer.off('jsonloaded'); // remove 'jsonloaded' event.

  • on(key, func)

    イベントを登録します。
    funcに渡される引数の最後に、伝播を管理する為のメソッドを持つオブジェクトが追加されます。

    observer.on('jsonloaded', function(vars, e) {
    // write code.
    console.log(vars); // {cir: 'Cool is Right.'}

    /*
    e.preventDefault(); // 現在のインスタンスのイベント実行を停止する
    e.stopPropagation(); // イベント伝播を停止する

    return false; // イベントの実行もイベント伝播も停止する
    */
});

observer.fire('jsonloaded', {
    cir: 'Cool is Right.'
});

  • one(key, func)

    一度だけ実行されるイベントを登録します。
    登録されたイベントはfireメソッドで呼び出されます。

    var func = observer.one('jsonloaded', function(vars){
    // write code.
    console.log(vars); // {cir: 'Cool is Right.'}
});

observer.fire('jsonloaded', 'fire1');
observer.fire('jsonloaded', 'fire2');

  • only(event, [vars])

    onメソッドで登録されたイベントを実行します。イベントは伝播されません。

    observer.on('jsonloaded', function(vars) {
    // write code.
    console.log(vars); // {cir: 'Cool is Right.'}
});

observer.only('jsonloaded', {
    cir: 'Cool is Right.'
});

  • removeChild([instance])

    Observer、もしくはObserverを継承したクラスのインスタンスを子から解除します。解除された子はイベント伝播の対象から外れます。
    instanceを省略した場合、すべての子を解除します。

    var observer2 = new C.Observer();
observer.addChild(observer2);

// write code.

observer.removeChild(observer2);
