• make(text)

    CSSスプライト用のタグを生成します。

    var tag_1 = font.make(1); // <span class="font_1">&nbsp;</span>
var tag_abc = font.make(abc);
// <span class="font_a">&nbsp;</span><span class="font_b">&nbsp;</span><span class="font_c">&nbsp;</span>
var tag_white = font.make('A');
// <span class="font_white_A">&nbsp;</span>
