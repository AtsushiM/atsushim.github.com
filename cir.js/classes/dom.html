• $(selector)

    document.querySelectorのショートハンドです。

    var $ = dom.$;
var $p = $('p'); // HTML Element

  • $$(selector)

    document.querySelectorAllのショートハンドです。
    返り値はNodeListではなくArrayに変換されています。

    var $$ = dom.$$;
var $$p = $$('p'); // array

  • $child(selector, element)

    element.querySelectorのショートハンドです。

    var $child = dom.$child;
var $p = $child('p', document.body); // HTML Element

  • $$child(selector, element)

    element.querySelectorAllのショートハンドです。
    返り値はNodeListではなくArrayに変換されています。

    var $$child = dom.$$child;
var $$p = $$('p', document.body); // array

  • $id(selector)

    document.getElementByIdのショートハンドです。

    var $id = dom.$id;
var $sample = $id('sample'); // HTML Element

  • addClass(element, value)

    elementのclassにvalueを追加します。

    var $sample = dom.$('#sample');
el.addClass($sample, 'sample');

  • attr(element, property, [value])

    elementの属性の取得、設定を行います。

    var $img = dom.$('#img');
var src = dom.attr($img, 'src'); // src取得
dom.attr($img, 'src', '/img/test.png'); // src設定
dom.attr($img, {
    width: '100px',
    height: '100px'
}); // まとめて設定

  • after(element, addElement)

    elementの後ろにaddElementを追加します。

    var div = dom.create('div');
var $sample = dom.$('#sample');
dom.after($sample, div);

  • append(_parent, element)

    _parentに対してelementを追加します。

    var div = dom.create('div');
var $sample = dom.$('#sample');
dom.append($sample, div);

  • before(element, addElement)

    elementの前にaddElementを追加します。

    var div = dom.create('div');
var $sample = dom.$('#sample');
dom.before($sample, div);

  • computedStyle(element)

    elementに適用されているcssのスタイルを取得する。
    (style属性に適用されていないものも取得する)

    var $sample = dom.$('#sample');
var computed = dom.computedStyle($sample);
var width = computed.width;
var height = computed.getPropertyValue('height');

  • create(tagname, [attr])

    document.createElementのエイリアスです。

    var div = dom.create('div'); // div element
var p = dom.create('p', {
    width: 100,
    height: 100
}); // set attr

  • css(element, object)

    element.styleにobjectの内容を反映します。

    var $sample = dom.$('#sample');
dom.css($sample, {
    width: '100px',
    height: '200px'
});

  • delegate(element, clsname, eventname, func)

    element内のclassNameにclsnameを持つ要素のeventnameイベントにfuncを登録します。

    var $sample = dom.$('#sample');
dom.delegate($sample, 'cir', C.e.CLICK, function() {
    console.log('Cool is Right.');
});

  • hasClass(element, value)

    elementのclassにvalueがあるかBooleanで返します。

    var $sample = dom.$('#sample');
var check = dom.hasClass($sample, 'sample'); // boolean

  • hide(element)

    element.style.displayに'none'を指定します。

    var $sample = dom.$('#sample');
dom.hide($sample);

  • html(element, [text])

    element.innerHTMLを取得・設定します。

    var $sample = dom.$('#sample');
var html = dom.html($sample); // 取得
dom.html($sample, 'rewrite innerHTML.'); // 設定

  • insertBefore(_parent, element)

    _parent内の先頭にelementを追加します。

    var div = dom.create('div');
var $sample = dom.$('#sample');
dom.insertBefore($sample, div);

  • off(element, eventname, handler)

    element.removeEventListnerを実行します。

    var $sample = dom.$('#sample');
var func = function() {
    alert('sample click!');
};
dom.on($sample, C.e.CLICK, func); // 追加
dom.off($sample, C.e.CLICK, func); // 削除

  • on(element, eventname, handler)

    element.addEventListnerを実行します。

    var $sample = dom.$('#sample');
dom.on($sample, C.e.CLICK, function() {
    alert('sample click!');
});

  • parent(element)

    elementの親エレメントを取得します。

    var $parent = dom.$('#sample').parent();

  • show(element)

    element.style.displayに'block'を指定します。

    var $sample = dom.$('#sample');
dom.show($sample);

  • reflow([el])

    表示を再描画(リフロー)します。
    elが指定されなかった場合、document.body以下をリフローします。

    dom.reflow();

  • remove(element)

    elementの削除を行います。

    dom.remove(element); // 削除

  • removeAttr(element, property)

    elementの属性の削除行います。

    var $img = dom.$('#img');
dom.attr($img, 'witdh', '100%'); // 設定
dom.removeAttr($img, 'width'); // 削除

  • removeClass(element, value)

    elementのclassからvalueを削除します。

    var $sample = dom.$('#sample');
dom.removeClass($sample, 'sample');

  • toggleClass(element, value)

    elementのclassにvalueがある場合は削除を、
    無い場合は追加します。

    var $sample = dom.$('#sample');
dom.toggleClass($sample, 'sample'); // sampleクラスを追加
dom.toggleClass($sample, 'sample'); // sampleクラスを削除

  • val(element, [text])

    element.valueを取得・設定します。

    var $sample = dom.$('#sample');
var value = dom.val($sample); // 取得
dom.val($sample, 'rewrite value.'); // 設定
