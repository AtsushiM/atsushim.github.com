• get([key])

    値を取得します。

    localStorage.cir = '{"c":"Cool","i":"is","r":"Right"}';
var cir = storage.get('cir'); // {c: 'Cool', i: 'is', r: 'Right'}

cir = storage.get(); // まとめて取得
/*
    cir == {cir: {c: 'Cool', i: 'is', r: 'Right'}}
*/

  • remove(key)

    値を削除します。

    storage.set('cir', 'Cool is Right.'); // 追加
storage.remove('cir'); // 削除

var cir = storage.get('cir'); // null

  • reset()

    保存された値を全て削除します。
    namespaceオプションが指定されない場合はlocalStorage全体が削除されます。

    storage.set('txt', 'Cool is Right.');
storage.set('num', 123);

storage.reset();

var txt = storage.get('txt'); // null
var num = storage.get('num'); // null

  • set(key, value)

    値を保存します。
    保存される値はStringに変換されます。

    storage.set('cir', {
    c: 'Cool',
    i: 'is',
    r: 'Right'
}); // localStorage.cir = '{"c":"Cool","i":"is","r":"Right"}';
