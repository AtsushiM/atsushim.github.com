• getHash()

    location.hashから値を取得します。

    location.hash = 'hashName?cir=Cool%20is%20Right.&id=123';
var hashobj = hashquery.getHash();
/*
{
    mode: 'hashName',
    contents: {
        cir: 'Cool is Right.',
        id: 123
    }
}
*/

  • makeHash({mode: string, contents: {key: val}})

    location.hashに設定するための値を作成します。

    var hashtxt = hashquery.makeHash({
    mode: 'hashName',
    contents: {
        cir: 'Cool is Right.',
        id: 123
    }
}); // hashName?cir=Cool%20is%20Right.&id=123

  • parseHash(querystring)

    クエリーストリング形式の文字列をオブジェクトに変換します。

    var hashvar = hashquery.parseHash('#hashName?cir=Cool%20is%20Right.&id=123');
/*
{
    mode: 'hashName',
    contents: {
        cir: 'Cool is Right.',
        id: 123
    }
}
*/

  • setHash({mode: string, contents: {key: val}})

    location.hashに値を設定します。

    var hashtxt = hashquery.setHash({
    mode: 'hashName',
    contents: {
        cir: 'Cool is Right.',
        id: 123
    }
}); // location.hash = '#hashName?cir=Cool%20is%20Right.&id=123'
