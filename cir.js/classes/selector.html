• .addClass(className)

    セレクターで指定されたHTMLエレメントのclassNameにvalueを追加します。

    var $sample = $('#sample');
$sample.addClass('sample');

  • .after(element)

    セレクターで指定されたHTMLエレメントの後ろにelementを追加します。

    var $sample = $('#sample');
var div = C.dom.create('div');
$sample.after(div);

  • .animate(params, [duration], [ease], [callback])

    セレクターで指定されたHTMLエレメントをアニメーションします。
    easeにはC.ease、もしくはC.sseaseの各プロパティ名を指定できます。

    var $sample = $('#sample');

$sample.animate({
    width: 100
}); // widthを100pxに変更

$sample.animate({
    height: '50%'
}, 1000); // 1000msでheightを50%に変更

$sample.animate({
    opacity: 0.5
}, 300, 'inCubic'); // 300msでopacityを0.5に'easeInCubic'でイージング

$sample.animate({
    opacity: 0.5
}, function() {
    // write code.
}); // durationとeaseは省略可能

  • .append(element)

    セレクターで指定されたHTMLエレメントにHTMLエレメントを追加します。

    var $sample = $('#sample');
var div = C.dom.create('div');
$sample.append(div);

  • .attr(property, [value])

    セレクターで指定されたHTMLエレメントの属性を取得：設定します。

    var $img = $('img');
var src = $img.attr('src'); // 取得
$img.attr('src', '/img/test.png'); // 設定
$img.attr({
    width: 100,
    height: 200
}); // まとめて設定

  • .before(element)

    セレクターで指定されたHTMLエレメントの前にelementを追加します。

    var $sample = $('#sample');
var div = C.dom.create('div');
$sample.before(div);

  • .css(object)

    セレクターで指定されたHTMLエレメントにまとめてstyleを適用します。

    var $sample = $('#sample');
$sample.css({
    width: 100,
    height: 100,
    paddingTop: 10
});

  • .delegate(clsname, eventname, func)

    セレクターで指定されたHTMLエレメント内のclassNameにclsnameを持つ要素のeventnameイベントにfuncを登録します。

    $('#sample').delegate('cir', C.e.CLICK, function() {
    console.log('Cool is Right.');
});

  • .find(selector)

    セレクターオブジェクトから子要素を取得します。

    var $sample = $('#sample');
var $sample_child = $sample.find('.sample');

  • .hasClass(className)

    セレクターで指定されたHTMLエレメントのclassNameにvalueが存在するかBooleanで返します。

    var $sample = $('#sample');
var has = $sample.hasClass('.sample'); // Boolean

  • .hide()

    セレクターで指定されたHTMLエレメントのdisplayに'none'を指定します。

    var $sample = $('#sample');
$sample.hide();

  • .html(text)

    セレクターで指定されたHTMLエレメントのinnerHTMLを取得・設定します。

    var $sample = $('#sample');
var html = $sample.html(); // 取得
$sample.html('Cool is Right.'); // 設定

  • .insertBefore(element)

    セレクターで指定されたHTMLエレメントの先頭にelementを追加します。

    var $sample = $('#sample');
var div = C.dom.create('div');
$sample.insertBefore(div);

  • .off(eventname, handler)

    セレクターで指定されたHTMLエレメントに対して、removeEventListnerを実行します。

    var $sample = $('#sample');
var handler = function() {
    alert('click!');
    // this = HTML element
};
$sample.on(C.e.CLICK, handler);
$sample.off(C.e.CLICK, handler);

  • .on(eventname, handler)

    セレクターで指定されたHTMLエレメントに対して、addEventListnerを実行します。

    var $sample = $('#sample');
$sample.on(C.e.CLICK, function(){
    alert('click!');
    // this = HTML element
});

  • .parent()

    セレクターオブジェクトの親要素を取得します。

    var $sample = $('#sample');
var $sample_parent = $sample.parent();

  • .querySelectorAll(query)

    セレクターオブジェクトの対象になっているHTMLエレメントに対して、element.querySelectorAllを実行します。
    このメソッドは他の$.methodから呼ばれるための関数であり、意図的に呼びだす必要はないはずです。

    var $sample = $('#sample');
var nodelist = $sample.querySelectorAll('.sample');

  • .remove()

    セレクターで指定されたHTMLエレメントを削除します。

    var $sample = $('#sample');
$smaple.remove(); // 削除

  • .removeAttr(property)

    セレクターで指定されたHTMLエレメントの属性を取得：設定します。

    var $img = $('img');
$img.attr('width', '100%'); // 設定
$img.removeAttr('width'); // 削除

  • .removeClass(className)

    セレクターで指定されたHTMLエレメントのclassNameからvalueを削除します。

    var $sample = $('#sample');
$sample.removeClass('sample');

  • .show()

    セレクターで指定されたHTMLエレメントのdisplayに'block'を指定します。

    var $sample = $('#sample');
$sample.show();

  • .stop()

    animateで指定されたアニメーションを停止します。
    animateの実行情報は各セレクターオブジェクト毎に保持されるため、
    変数にセレクターオブジェクトを保存していない場合、正しく停止できない可能性があります。

    var $sample = $('#sample');

$sample.animate({
    width: 100
}); // widthを100pxに変更

$sample.stop(); // 停止する

$('#sample2').animate({
    height: 200
}); // heightを200pxに変更
$('#sample2').stop(); // 停止しない

  • .toggleClass(className)

    セレクターで指定されたHTMLエレメントに対して
    ・classNameにvalueがない場合は追加
    ・classNameにvalueがある場合は削除
    を実行します。

    var $sample = $('#sample');
$sample.toggleClass('sample'); // 追加
$sample.toggleClass('sample'); // 削除

  • .undelegate(clsname, eventname, [func])

    delegateメソッドで登録されたイベントを削除します。
    funcが省略された場合、clsname、eventnameで登録されたすべてのイベントが削除されます。

    var $sample = $('#sample');
$sample.delegate('cir', C.e.CLICK, function() {
    console.log('Cool is Right.');
});
$sample.undelegate('cir', C.e.CLICK);

  • .val(text)

    セレクターで指定されたHTMLエレメントのvalueを取得・設定します。

    var $sample = $('#sample');
var value = $sample.val(); // 取得
$sample.val('Cool is Right.'); // 設定
