CIR.js

Cool is Right.

CIR.jsはスマートフォンメインのWeb&ハイブリッドアプリ開発用のライブラリです。
ほとんどの機能は使わない場合排除可能で、カスタマイズできます。

ダウンロード：GitHub

C
var ajax = new C.Ajax(); // ajax
var mobile = new C.Mobile(); // Mobile

CIR.jsのネームスペースです。
すべてのクラス、メソッドはC以下に存在します。

C.utility
var util = C.utility;

様々なクラスから呼び出されるユーティリティオブジェクトです。

  • pageTop()

    ページ上部へスクロールを移動させます。
    iPhoneなどのスマートフォンの場合、スクロールバーを表示しません。

    util.pageTop();

  • override(targetObj, varObj)

    targetObjに対してvarObjのプロパティを上書きします。
    返り値は上書きされたtargetObjになります。

    var target = {};
var over = {sample: 'override property'};
util.override(target, over);
alert(target.sample); // override property

  • replaceAll(targetext, needle, replacetext)

    targettext内にあるneedleを全てreplacetextに書き換えます。
    正規表現は使えません。

    var text = util.replaceAll('cir.js', 'cir', 'Cool is Right');
alert(text); // Cool is Right.js

text = util.relaceAll(text, 's', 'a');
alert(text); // Cool ia Right.ja

  • windowOpen(url, windowname)

    新しいウィンドウを開きます

    var subwindow = util.windowOpen('/sub.html', 'sub');

  • typeCast(string)

    渡された文字列を内容に合わせて型変換します。

    var obj = '{"a":"objtext"}';
var num = '123';
var boolTrue = 'true';
var boolFalse = 'false';
var str = 'sample';

obj = util.typeCast(obj); // { a: 'objtext'}
num = util.typeCast(num); // 123
boolTrue = util.typeCast(boolTrue); // true
boolFalse = util.typeCast(boolFalse); // false
str = util.typeCast(str); // 'sample'

  • makeQueryString(object)

    オブジェクトのプロパティをlocation.search、もしくはlocation.hash用に使えるクエリーストリング形式に変換します。

    var hashquery = util.makeQueryString({
    c: 'Cool',
    i: 'is',
    r: 'Right'
}); // c=Cool&i=is&r=Right

  • parseQueryString(string)

    location.search、もしくはlocation.hashの値をオブジェクトに変換します。

    var hashquery = util.makeQueryString({
    c: 'Cool',
    i: 'is',
    r: 'Right'
}); // c=Cool&i=is&r=Right
location.hash = hashquery;

var obj = util.parseQueryString(location.hash);
alert(obj.c + ' ' + obj.i + ' ' + obj.r + '!');

  • isObject(vars)

    varsがObjectかBooleanで返します。

    var check;
check = util.isObject({}); // true
check = util.isObjext(''); // false

  • isNumber(vars)

    varsがNumberかBooleanで返します。

    var check;
check = util.isNumber(123); // true
check = util.isNumber('123'); // false

  • isString(vars)

    varsがStringかBooleanで返します。

    var check;
check = util.isString(''); // true
check = util.isString(12); // false

  • isFunction(vars)

    varsがFunctionかBooleanで返します。

    var check;
check = util.isFunction((function() {})); // true
check = util.isFunction(12); // false

  • isBoolean(vars)

    varsがBooleanかBooleanで返します。

    var check;
check = util.isBoolean(true); // true
check = util.isBoolean(false); // true
check = util.isBoolean(1); // false

  • is(type, vars)

    varsがtypeで指定された形式になっているかBooleanで返します。

    var check;
check = util.is('String', ''); // true
check = util.is('Number', '123'); // false

  • nullFunction()

    nullを返却するメソッドです。
    実質的に何もしませんが、メソッドの引数で渡すcallbackが必須でない場合等で使うと効果的です。

    function anyAction(callback) {
    // any action...
    callback();
}
anyAction(util.nullFunction); // anyActionは常にcallbackを必要とするのでnullFunctionを渡し処理の簡略
C.element
var el = C.element;

HTMLエレメントに関するメソッドをまとめたオブジェクトです。

  • $(selector)

    document.querySelectorのショートハンドです。

    var $ = el.$;
var $p = $('p'); // HTML Element

  • $$(selector)

    document.querySelectorAllのショートハンドです。
    返り値はNodeListではなくArrayに変換されています。

    var $$ = el.$$;
var $$p = $$('p'); // array

  • $child(selector, element)

    element.querySelectorのショートハンドです。

    var $child = el.$child;
var $p = $child('p', document.body); // HTML Element

  • $$child(selector, element)

    element.querySelectorAllのショートハンドです。
    返り値はNodeListではなくArrayに変換されています。

    var $$child = el.$$child;
var $$p = $$('p', document.body); // array

  • $id(selector)

    document.getElementByIdのショートハンドです。

    var $id = el.$id;
var $sample = $id('sample'); // HTML Element

  • on(element, eventname, handler)

    element.addEventListnerを実行します。

    var $sample = el.$('#sample');
el.on($sample, C.event.click, function() {
    alert('sample click!');
});

  • off(element, eventname, handler)

    element.removeEventListnerを実行します。

    var $sample = el.$('#sample');
var func = function() {
    alert('sample click!');
};
el.on($sample, C.event.click, func);
el.off($sample, C.event.click, func);

  • create(tagname [, attr])

    document.createElementのエイリアスです。

    var div = el.create('div'); // div element

  • show(element)

    element.style.displayに'block'を指定します。

    var $sample = el.$('#sample');
el.show($sample);

  • hide(element)

    element.style.displayに'none'を指定します。

    var $sample = el.$('#sample');
el.hide($sample);

  • opacity(element, value)

    element.style.opacityにvalueを指定します。

    var $sample = el.$('#sample');
el.opacity($sample, 0.5);

  • hasClass(element, value)

    elementのclassにvalueがあるかBooleanで返します。

    var $sample = el.$('#sample');
el.hasClass($sample, 'sample');

  • addClass(element, value)

    elementのclassにvalueを追加します。

    var $sample = el.$('#sample');
el.addClass($sample, 'sample');

  • removeClass(element, value)

    elementのclassからvalueを削除します。

    var $sample = el.$('#sample');
el.removeClass($sample, 'sample');

  • toggleClass(element, value)

    elementのclassにvalueがある場合は削除を、
    無い場合は追加します。

    var $sample = el.$('#sample');
el.toggleClass($sample, 'sample'); // sampleクラスを追加
el.toggleClass($sample, 'sample'); // sampleクラスを削除

  • style(element, object)

    element.styleにobjectの内容を反映します。

    var $sample = el.$('#sample');
el.style({
    width: '100px',
    height: '200px'
});

  • computedStyle(element)

    elementに適用されているcssのスタイルを取得する。
    (style属性に適用されていないものも取得する)

    var $sample = el.$('#sample');
var computed = el.computedStyle($sample);
var width = computed.width;
var height = computed.getPropertyValue('height');

  • attr(element, property, value)

    elementの属性の取得、設定を行います。

    var $img = el.$('#img');
var src = el.attrElement($img, 'src'); // src取得
el.attr($img, 'src', '/img/test.png'); // src設定
el.attr($img, {
    width: '100px',
    height: '100px'
}); // まとめて設定

  • append(_parent, element)

    _parentに対してelementを追加します。

    var div = el.create('div');
var $sample = el.$('#sample');
el.append($sample, div);

  • html(element, text)

    element.innerHTMLを取得・設定します。

    var $sample = el.$('#sample');
el.innerHTML($sample); // 取得
el.innerHTML($sample, 'rewrite innerHTML.'); // 設定

C.selector
var $ = C.selector;
var $p = $('p');

jQueryライクなセレクターメソッドです。
戻り値は便宜上セレクターオブジェクトとします。

  • .querySelectorAll(query)

    セレクターオブジェクトの対象になっているHTMLエレメントに対して、element.querySelectorAllを実行します。
    このメソッドは他のselector.methodから呼ばれるための関数であり、意図的に呼びだす必要はないはずです。

    var $sample = $('#sample');
var nodelist = $sample.querySelectorAll('.sample');

  • .find(selector)

    セレクターオブジェクトから子要素を取得します。

    var $sample = $('#sample');
var $sample_child = $sample.find('.sample');

  • .parent()

    セレクターオブジェクトの親要素を取得します。

    var $sample = $('#sample');
var $sample_parent = $sample.parent();

  • .on(eventname, handler)

    セレクターで指定されたHTMLエレメントに対して、addEventListnerを実行します。

    var $sample = $('#sample');
$sample.on(C.event.click, function(){
    alert('click!');
    // this = HTML element
});

  • .off(eventname, handler)

    セレクターで指定されたHTMLエレメントに対して、removeEventListnerを実行します。

    var $sample = $('#sample');
var handler = function() {
    alert('click!');
    // this = HTML element
};
$sample.on(C.event.click, handler);

// onで設定したものを外す
$sample.off(C.event.click, handler);

  • .show()

    セレクターで指定されたHTMLエレメントのdisplayに'block'を指定します。

    var $sample = $('#sample');
$sample.show();

  • .hide()

    セレクターで指定されたHTMLエレメントのdisplayに'none'を指定します。

    var $sample = $('#sample');
$sample.hide();

  • .opacity(value)

    セレクターで指定されたHTMLエレメントのopacityを設定します。

    var $sample = $('#sample');
$sample.opacity(0.5);

  • .hasClass(value)

    セレクターで指定されたHTMLエレメントのclassNameにvalueが存在するかBooleanで返します。

    var $sample = $('#sample');
var has = $sample.hasClass('.sample'); // Boolean

  • .addClass(value)

    セレクターで指定されたHTMLエレメントのclassNameにvalueを追加します。

    var $sample = $('#sample');
$sample.addClass('sample');

  • .removeClass(value)

    セレクターで指定されたHTMLエレメントのclassNameからvalueを削除します。

    var $sample = $('#sample');
$sample.removeClass('sample');

  • .toggleClass(value)

    セレクターで指定されたHTMLエレメントに対して
    ・classNameにvalueがない場合は追加
    ・classNameにvalueがある場合は削除
    を実行します。

    var $sample = $('#sample');
$sample.toggleClass('sample'); // 追加
$sample.toggleClass('sample'); // 削除

  • .css(object)

    セレクターで指定されたHTMLエレメントにまとめてstyleを適用します。

    var $sample = $('#sample');
$sample.css({
    width: 100,
    height: 100,
    paddingTop: 10
});

  • .append(value)

    セレクターで指定されたHTMLエレメントにHTMLエレメントを追加します。

    var $sample = $('#sample');
var div = C.element.create('div');
$sample.append(div);

  • .html(value)

    セレクターで指定されたHTMLエレメントのinnerHTMLを取得・設定します。

    var $sample = $('#sample');
var html = $sample.html(); // 取得
$sample.html('Cool is Right.'); // 設定

  • .attr(value)

    セレクターで指定されたHTMLエレメントの属性を取得：設定します。

    var $img = $('img');
var src = $img.attr('src'); // 取得
$img.attr('src', '/img/test.png'); // 設定
$img.attr({
    width: 100,
    height: 200
}); // まとめて設定

  • .animate(value)

    セレクターで指定されたHTMLエレメントをアニメーションします。

    var $sample = $('#sample');

$sample.animate({
    width: 100
}); // widthを100pxに変更

$sample.animate({
    height: '50%'
}, 1000); // 1000msでheightを50%に変更

$sample.animate({
        opacity: 0.5
    }, 300, {
        easing: C.easing.easeInCubic,
        onComplete: function() {
            // write code.
        }
}); // 300msでopacityを0.5にイージング処理し、コールバックを実行
